Script Python utilisé pour extraire l'intensité du canal vert et calculer la concentration en anticorps.

analyse_lfa.py 
from PIL import Image
import numpy as np
import matplotlib.pyplot as plt
from matplotlib.widgets import RectangleSelector
import tkinter as tk
from tkinter import filedialog
import sys

# ---------------------------------------------------
# SÉLECTION DU FICHIER
# ---------------------------------------------------

root = tk.Tk()
root.withdraw()

print("📂 Ouverture de la fenêtre de sélection de fichier...")

IMAGE_PATH = filedialog.askopenfilename(
    title="Sélectionnez l'image du test",
    filetypes=[("Images", "*.jpg *.jpeg *.png *.bmp *.tif")]
)

if not IMAGE_PATH:
    print("❌ Aucun fichier sélectionné. Arrêt du programme.")
    sys.exit()

print(f"✅ Fichier chargé : {IMAGE_PATH}")

# ---------------------------------------------------
# PARAMÈTRES DE CALIBRATION
# ---------------------------------------------------

a = 8.12      # pente de la droite
b = 152.33    # ordonnée à l'origine

roi_coords = None

# ---------------------------------------------------
# FONCTIONS
# ---------------------------------------------------

def onselect(eclick, erelease):
    """Callback quand l'utilisateur termine la sélection."""
    global roi_coords
    x1, y1 = int(eclick.xdata), int(eclick.ydata)
    x2, y2 = int(erelease.xdata), int(erelease.ydata)
    roi_coords = (min(x1, x2), min(y1, y2),
                  max(x1, x2), max(y1, y2))
    print("\n📌 Zone sélectionnée :", roi_coords)


def compute_green_intensity(roi):
    """Calcule l'intensité moyenne du VERT (canal G)."""
    green = roi[:, :, 1].astype(np.float32)
    return float(np.mean(green))


def calibration(intensity, a, b):
    """Applique la droite de calibration."""
    return (intensity - b) / a

# ---------------------------------------------------
# PROGRAMME PRINCIPAL
# ---------------------------------------------------

if __name__ == "__main__":

    img = Image.open(IMAGE_PATH).convert("RGB")
    arr = np.array(img)

    print("➡️  Sélectionnez la zone, puis fermez la fenêtre.")

    fig, ax = plt.subplots(figsize=(8, 6))
    ax.imshow(arr)
    ax.set_title("Sélectionnez la zone puis fermez la fenêtre")

    rect_selector = RectangleSelector(
        ax, onselect, useblit=True,
        button=[1], minspanx=5, minspany=5,
        spancoords='pixels', interactive=True
    )
    plt.show()

    if roi_coords is None:
        print("❌ Aucune zone sélectionnée.")
        sys.exit()

    x1, y1, x2, y2 = roi_coords
    h, w = arr.shape[:2]
    x1, x2 = max(0, min(w-1, x1)), max(0, min(w-1, x2))
    y1, y2 = max(0, min(h-1, y1)), max(0, min(h-1, y2))
    roi = arr[y1:y2, x1:x2]

    green_intensity = compute_green_intensity(roi)
    concentration = calibration(green_intensity, a, b)

    print("\n🟩 Intensité moyenne du vert =", green_intensity)
    print("📘 Concentration déduite =", concentration)
